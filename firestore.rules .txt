rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {

    // ── HELPER : est-ce l'admin ? ─────────────────────────────────
    function isAdmin() {
      return request.auth != null &&
        exists(/databases/$(database)/documents/system/config) &&
        get(/databases/$(database)/documents/system/config).data.adminUid == request.auth.uid;
    }

    // ── HELPER : champs écrits par l'admin pour les badges ─────────
    function isBadgeUpdate() {
      return request.resource.data.diff(resource.data).affectedKeys()
        .hasOnly([
          'isVerified', 'verifiedUntil', 'verifiedByAdmin', 'verifiedAt',
          'verificationPending',
          'isPremium', 'premiumUntil', 'premiumByAdmin', 'premiumAt',
          'tier',
          'isBanned', 'banReason', 'banAt', 'banByAdmin',
          'riskLevel', 'riskReportCount',
          'advancePaymentAllowed'
        ]);
    }

    // ── PRODUITS ─────────────────────────────────────────────────
    match /products/{productId} {
      allow read: if true;
      allow create: if request.auth != null;
      allow update: if request.auth != null && (
        request.resource.data.diff(resource.data).affectedKeys()
          .hasAny(['viewCount', 'whatsappClickCount'])
        && request.resource.data.diff(resource.data).affectedKeys()
          .hasOnly(['viewCount', 'whatsappClickCount',
            'viewsByHour', 'viewsByWeek', 'bookmarkCount',
            'flashSaleActive', 'delivererAccepted', 'delivererEnRoute', 'delivererGpsLink'])
      );
      allow update: if request.auth != null && (
        request.auth.uid == resource.data.sellerId || isAdmin()
      );
      allow delete: if request.auth != null && (
        request.auth.uid == resource.data.sellerId || isAdmin()
      );
    }

    match /dettes/{id} {
      allow read, write: if request.auth != null &&
        (request.auth.uid == resource.data.userId ||
         request.auth.uid == request.resource.data.userId);
    }

    match /compta/{entryId} {
      allow read, write: if request.auth != null &&
        (request.auth.uid == resource.data.userId ||
         request.auth.uid == request.resource.data.userId);
    }

    match /clients_carnet/{id} {
      allow read, write: if request.auth != null &&
        (request.auth.uid == resource.data.userId ||
         request.auth.uid == request.resource.data.userId);
    }

    // ── UTILISATEURS ─────────────────────────────────────────────
    match /users/{userId} {
      allow read: if request.auth != null;
      allow create: if request.auth != null && request.auth.uid == userId;
      allow update: if request.auth != null && (
        // Propriétaire modifie son propre profil
        request.auth.uid == userId
        ||
        // Admin peut tout modifier
        isAdmin()
        ||
        // Mise à jour parrainage par le service
        request.resource.data.diff(resource.data).affectedKeys()
          .hasOnly(['referralCount', 'referralBonusPublications',
                    'referralBonusChats', 'referralFreeVerifiedUntil', 'isVerified'])
        ||
        // Trust system — mise à jour riskLevel
        request.resource.data.diff(resource.data).affectedKeys()
          .hasOnly(['riskLevel', 'riskReportCount', 'isBanned'])
        ||
        // Stats livreur (incrément par livreur sur son propre profil ou par le système)
        request.resource.data.diff(resource.data).affectedKeys()
          .hasOnly(['totalDeliveries', 'totalEarnings', 'delivererReferralCount',
                    'delivererReferralBonusAt', 'delivererAccepted'])
        ||
        // Présence & activité (chaque user sur son propre profil)
        request.resource.data.diff(resource.data).affectedKeys()
          .hasOnly(['lastActiveAt', 'avgResponseTime'])
        ||
        // Cashback fidélité (acheteur sur son propre profil)
        request.resource.data.diff(resource.data).affectedKeys()
          .hasOnly(['loyaltyPoints', 'loyaltyPointsUsed'])
        ||
        // Suivre/ne plus suivre un vendeur (acheteur sur son propre profil)
        request.resource.data.diff(resource.data).affectedKeys()
          .hasOnly(['followingSellers'])
        ||
        // Wishlist (acheteur sur son propre profil)
        request.resource.data.diff(resource.data).affectedKeys()
          .hasOnly(['wishlistIds', 'wishlistPublic', 'wishlistSlug'])
        ||
        // Admin badge updates (Vérifié + Premium + ban)
        isBadgeUpdate()
      );
    }

    // ── TRUST REPORTS ─────────────────────────────────────────────
    match /trust_reports/{reportId} {
      allow create: if request.auth != null
        && request.resource.data.reporterId == request.auth.uid
        && request.resource.data.details.size() >= 20
        && request.resource.data.status == 'pending';
      allow read: if request.auth != null
        && (isAdmin() || resource.data.reporterId == request.auth.uid);
      allow list: if request.auth != null;
      allow update: if request.auth != null && isAdmin();
      allow delete: if false;
    }

    // ── TRUST SCORES ──────────────────────────────────────────────
    match /trust_scores/{userId} {
      allow read: if request.auth != null;
      allow create, update: if request.auth != null;
      allow delete: if false;
    }

    // ── CONVERSATIONS / MESSAGES ──────────────────────────────────
    match /conversations/{convId} {
      allow read: if request.auth != null && (
        request.auth.uid in resource.data.participants || isAdmin()
      );
      allow create: if request.auth != null;
      allow update: if request.auth != null && (
        request.auth.uid in resource.data.participants || isAdmin()
      );
      match /messages/{msgId} {
        allow read, write: if request.auth != null;
      }
    }

    // ── AVIS / REVIEWS ────────────────────────────────────────────
    match /reviews/{reviewId} {
      allow read: if true;
      allow create: if request.auth != null;
      allow update, delete: if request.auth != null && (
        request.auth.uid == resource.data.reviewerId || isAdmin()
      );
    }

    // ── NOTIFICATIONS ─────────────────────────────────────────────
    match /notifications/{userId} {
      allow read, write: if request.auth != null && (
        request.auth.uid == userId || isAdmin()
      );
      match /items/{notifId} {
        allow read, write: if request.auth != null && (
          request.auth.uid == userId || isAdmin()
        );
        allow create: if request.auth != null;
      }
    }

    // ── REFERRALS (parrainage livreur) ─────────────────────────────
    match /referrals/{doc} {
      allow read: if request.auth != null && (
        resource.data.ownerId == request.auth.uid || isAdmin()
      );
      allow create: if request.auth != null;
      allow update, delete: if isAdmin();
    }

    // ── COMMANDES / ORDERS ────────────────────────────────────────
    match /orders/{orderId} {
      allow read: if request.auth != null && (
        request.auth.uid == resource.data.buyerId ||
        request.auth.uid == resource.data.sellerId ||
        request.auth.uid == resource.data.delivererId ||
        isAdmin()
      );
      allow create: if request.auth != null;
      allow update: if request.auth != null && (
        request.auth.uid == resource.data.buyerId ||
        request.auth.uid == resource.data.sellerId ||
        request.auth.uid == resource.data.delivererId ||
        isAdmin()
      );
    }

    // ── STORIES ──────────────────────────────────────────────────
    // Lecture : tout le monde (accueil public)
    // Création : vendeur Vérifié OU Premium (isVerified OR isPremium sur le user)
    match /stories/{storyId} {
      allow read: if true;
      allow create: if request.auth != null
        && request.resource.data.sellerId == request.auth.uid
        && (
          get(/databases/$(database)/documents/users/$(request.auth.uid)).data.isVerified == true
          || get(/databases/$(database)/documents/users/$(request.auth.uid)).data.isPremium == true
        );
      allow update: if request.auth != null && (
        request.auth.uid == resource.data.sellerId
        || request.resource.data.diff(resource.data).affectedKeys().hasOnly(['views'])
        || isAdmin()
      );
      allow delete: if request.auth != null && (
        request.auth.uid == resource.data.sellerId || isAdmin()
      );
    }

    // ── BOOSTS ───────────────────────────────────────────────────
    match /boosts/{boostId} {
      allow read: if true;
      allow create: if request.auth != null
        && request.resource.data.sellerId == request.auth.uid;
      allow update: if request.auth != null && (
        resource.data.sellerId == request.auth.uid || isAdmin()
      );
      allow delete: if request.auth != null && (
        resource.data.sellerId == request.auth.uid || isAdmin()
      );
    }

    // ── ADMIN LOGS ───────────────────────────────────────────────
    match /admin_logs/{logId} {
      allow read: if isAdmin();
      allow create: if isAdmin();
      allow update, delete: if false;
    }

    // ── SYSTEM ───────────────────────────────────────────────────
    match /system/{docId} {
      allow read: if true;
      allow write: if isAdmin();
    }

    // ── SYSTEM BANNERS ───────────────────────────────────────────
    match /system_banners/{bannerId} {
      allow read: if true;
      allow write: if isAdmin();
    }

    // ── PARRAINAGE ───────────────────────────────────────────────
    match /referrals/{refId} {
      allow read: if request.auth != null;
      allow create: if request.auth != null;
      allow update: if request.auth != null && (
        request.auth.uid == resource.data.ownerId || isAdmin()
      );
      allow delete: if request.auth != null && (
        request.auth.uid == resource.data.ownerId || isAdmin()
      );
    }

    // ── SIGNALEMENTS ─────────────────────────────────────────────
    match /reports/{reportId} {
      allow create: if request.auth != null;
      allow read: if isAdmin();
      allow update: if isAdmin();
      allow delete: if false;
    }

    // ── ALERTES DE RECHERCHE ─────────────────────────────────────
    match /search_alerts/{alertId} {
      allow read: if request.auth != null &&
        (resource.data.userId == request.auth.uid || isAdmin());
      allow create: if request.auth != null &&
        request.resource.data.userId == request.auth.uid;
      allow delete: if request.auth != null &&
        resource.data.userId == request.auth.uid;
      allow update: if false;
    }

    // ── BOOKMARKS ────────────────────────────────────────────────
    match /bookmarks/{bookmarkId} {
      allow read: if request.auth != null
        && resource.data.userId == request.auth.uid;
      allow create: if request.auth != null
        && request.resource.data.userId == request.auth.uid;
      allow delete: if request.auth != null
        && resource.data.userId == request.auth.uid;
      allow update: if false;
    }

    // ── APP CONFIG ───────────────────────────────────────────────
    match /appConfig/{docId} {
      allow read: if true;
      allow write: if isAdmin();
    }

    // ── AUTH TOKENS ──────────────────────────────────────────────
    match /authTokens/{tokenId} {
      allow read: if true;
      allow write: if false;
    }

    // ── PUSH SUBSCRIPTIONS ───────────────────────────────────────
    match /push_subscriptions/{userId} {
      allow read, write: if request.auth != null && request.auth.uid == userId;
    }

    // ── CATCH-ALL ─────────────────────────────────────────────────
    match /{document=**} {
      allow read, write: if false;
    }
  }
}
